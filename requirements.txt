FROM python:3.11

WORKDIR /app

# Copy chung requirements
COPY ../../requirements.txt ./requirements.txt

# Cài system packages cần thiết
RUN apt-get update && \
    apt-get install -y gcc libpq-dev && \
    rm -rf /var/lib/apt/lists/*

# Install packages 1 lần cho tất cả
RUN pip install --no-cache-dir -r requirements.txt

# Copy code riêng service
COPY . .

CMD ["python", "ingestion.py"]  # hoặc uvicorn main:app ...
